ડેટા બ્રાઉઝર માં સેવ થાય છે!
આ App નો બધો ડેટા Browser (Local Storage) માં સેવ થાય છે.
Browser History Clear, Cache Clear અથવા Browser Uninstall કરવાથી ડેટા કાયમ માટે ડિલીટ થઈ જશે.
તેથી નિયમિત Backup લેવો ખૂબ જ જરૂરી છે!
1
💾 Backup કેવી રીતે લેવો?
બધો ડેટા JSON ફાઈલ (Backup) તરીકે Download કરો અને Safe જગ્યા (Google Drive / Pendrive) માં save કરો.
Backup ક્યાં મળશે?
📍 Sidebar → Shop & System Section → "Exit" button પર click કરો
📍 Exit modal માં "💾 બેકઅપ લો અને બહાર નીકળો" button click કરો
📍 ઓટો-બેકઅપ Folder set કરો → Sidebar → "📁 ઓટો-બેકઅપ સેટ કરો"
    (folder select કર્યા પછી Exit વખતે ઓટોમેટિક save થશે)
2
📂 Restore (ડેટા પાછો લાવો) કેવી રીતે?
ડેટા ડિલીટ થઈ ગયો હોય અથવા નવા Device/Browser માં App ખોલો ત્યારે Backup ફાઈલ Upload કરો.
Restore ક્યાં મળશે?
📍 App ખોલો → Welcome Screen → "📂 જૂનો ડેટા અપલોડ કરો (Restore)" button
📍 અથવા Sidebar → "📂 Data રીસ્ટોર (Restore)" → Backup JSON ફાઈલ Select કરો
📍 Restore પછી બધો ડેટા (Bills, Products, Customers, Repairs) પાછો આવી જશે

// ===== VyaparApp Google Sheets Script v4 =====

function getSheet(name) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function writeSheet(name, rows) {
  if (!rows || rows.length === 0) return;
  var sheet = getSheet(name);
  sheet.clearContents();
  sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
  sheet.getRange(1, 1, 1, rows[0].length)
    .setBackground('#1a56db').setFontColor('#ffffff').setFontWeight('bold');
}

function saveJson(jsonStr) {
  var props = PropertiesService.getScriptProperties();
  // Split into 9KB chunks (Apps Script property limit is 9KB per value)
  var CHUNK = 9000;
  var i = 0, idx = 0;
  // Clear old keys first
  var old = props.getKeys().filter(function(k){ return k.indexOf('vdb_') === 0; });
  old.forEach(function(k){ props.deleteProperty(k); });
  // Save new chunks
  while (i < jsonStr.length) {
    props.setProperty('vdb_' + idx, jsonStr.substring(i, i + CHUNK));
    i += CHUNK;
    idx++;
  }
  props.setProperty('vdb_count', String(idx));
}

function readJson() {
  var props = PropertiesService.getScriptProperties();
  var count = parseInt(props.getProperty('vdb_count') || '0');
  if (count === 0) return null;
  var parts = [];
  for (var i = 0; i < count; i++) {
    parts.push(props.getProperty('vdb_' + i) || '');
  }
  return parts.join('');
}

function processPayload(payload) {
  if (payload.action === 'save_all') {
    saveJson(payload.json);
    if (payload.bills)     writeSheet('Bills',     payload.bills);
    if (payload.customers) writeSheet('Customers', payload.customers);
    if (payload.products)  writeSheet('Products',  payload.products);
    if (payload.expenses)  writeSheet('Expenses',  payload.expenses);
    if (payload.repairs)   writeSheet('Repairs',   payload.repairs);
    if (payload.purchases) writeSheet('Purchases', payload.purchases);
    return {status:'ok'};
  }
  if (payload.action === 'save_json') {
    saveJson(payload.json);
    return {status:'ok'};
  }
  if (payload.action === 'write_sheet') {
    writeSheet(payload.sheet, payload.rows);
    return {status:'ok', sheet: payload.sheet};
  }
  return {status:'error', msg:'Unknown action: ' + payload.action};
}

function doGet(e) {
  try {
    // Read request
    if (e.parameter.action === 'read') {
      var data = readJson();
      return out({status:'ok', data: data});
    }
    // GET with payload
    if (e.parameter.d) {
      var payload = JSON.parse(decodeURIComponent(e.parameter.d));
      return out(processPayload(payload));
    }
    return out({status:'ok', msg:'VyaparApp v4 ready'});
  } catch(err) {
    return out({status:'error', msg: err.toString()});
  }
}

function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    return out(processPayload(payload));
  } catch(err) {
    return out({status:'error', msg: err.toString()});
  }
}

function out(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
